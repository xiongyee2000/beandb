CMAKE_MINIMUM_REQUIRED(VERSION 2.8)

PROJECT(reactor CXX)

add_definitions(-g -std=c++11)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -g -Wall --coverage")
set(REACTOR_DEP_INC_PATH ${CMAKE_CURRENT_SOURCE_DIR}/../dependencies/include)
set(3RDPARTY_PATH ${CMAKE_CURRENT_SOURCE_DIR}/../dependencies/include/thirdParty/)
set(REACTOR_DEP_LIB_PATH ${CMAKE_CURRENT_SOURCE_DIR}/../dependencies/libs)
set(REACTOR_SRC_PATH ${CMAKE_CURRENT_SOURCE_DIR})

file(GLOB_RECURSE REACTOR_SRC_FILES "*.cpp")

INCLUDE_DIRECTORIES(${REACTOR_DEP_INC_PATH})
INCLUDE_DIRECTORIES(${3RDPARTY_PATH})
INCLUDE_DIRECTORIES(${ROADDB_CORE_INC_PATH})
INCLUDE_DIRECTORIES(${REACTOR_SRC_PATH})

LINK_DIRECTORIES(${REACTOR_DEP_LIB_PATH}/x64)

ADD_LIBRARY(reactor SHARED ${REACTOR_SRC_FILES})

#link libraries
set (REACTOR_LINK_LIBRARY
    pthread
)

set (REACTOR_SRV_LINK_LIBRARY
    pthread
)

TARGET_LINK_LIBRARIES(reactor ${REACTOR_LINK_LIBRARY})
